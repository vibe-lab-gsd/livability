Load data

City-level data assembled in data-assembly.

Code 
city_data <- here("final_fig_data.geojson") |>
  st_read() |>
  mutate(migr_pct_trunc = ifelse(migration_pct < -0.05, -0.05, migration_pct)) |>
  mutate(migr_pct_trunc = ifelse(migr_pct_trunc > 0.05, 0.05, migr_pct_trunc))

Create a basic basemap.

Code 
states_map <- states() |>
  filter(STUSPS %in% c("RI", 
                       "NJ",
                       "CT",
                       "IL",
                       "IN",
                       "MA",
                       "ME",
                       "MN",
                       "MI",
                       "ND",
                       "NH",
                       "NY",
                       "OH",
                       "PA",
                       "VT",
                       "WI")) 

great_lakes <- ne_download(type = "lakes", category = "physical", scale = 50) |>
  st_transform(crs = st_crs(states_map)) |>
  st_filter(states_map)

cape_cod <- area_water(state = "MA", county = c("Barnstable", 
                                                "Plymouth", 
                                                "Suffolk",
                                                "Nantucket",
                                                "Dukes"))

long_island <- area_water(state = "NY", county = c("Suffolk", 
                                                   "Nassau",
                                                   "Queens",
                                                   "Kings",
                                                   "New York"))

nj_water <- area_water(state = "NJ", county = c("Bergen", 
                                                "Hudson",
                                                "Union",
                                                "Middlesex",
                                                "Monmouth"))

crs_frostbelt = "+proj=lcc +lat_0=39 +lon_0=-89 +lat_1=37 +lat_2=47 +x_0=0 +y_0=0 +datum=NAD83 +units=m +no_defs +type=crs"

base_map <- ggplot(states_map) +
  geom_sf(fill = "gray90",
          color = "white") +
  geom_sf(data = great_lakes,
          color = NA, 
          fill = "white") +
  geom_sf(data = cape_cod, fill = "white", color = NA) +
  geom_sf(data = long_island, fill = "white", color = NA) +
  geom_sf(data = nj_water, fill = "white", color = NA)  +
  theme_map()

Clusters

We used k-means clustering to identify five types of frostbelt city. Here’s how the clusters map onto plan scores.

Code 
ggplot(city_data) +
  geom_violin(aes(x = cluster, y = plan_pts,
                  fill = cluster),
              alpha = 0.2) +
  geom_jitter(aes(x = cluster, y = plan_pts,
                  color = cluster),
              alpha = 0.2) +
  scale_fill_brewer(type = "qual", palette = 2) +
  scale_color_brewer(type = "qual", palette = 2) +
  scale_y_continuous(name = "Plan quality score") +
  theme_minimal()

Here’s how the clusters map onto population.

Code
